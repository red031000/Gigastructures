giga_tech_svalinn_installation = {
	cost = 200000
	area = society
	tier = 5
	category = { military_theory }
	is_rare = yes
	prerequisites = { "giga_tech_tetradimensional_engineering" "giga_tech_maginot_world" "giga_tech_interstellar_habitat" }
	weight = @tier5weight3
	prereqfor_desc = {

		#custom = { title = "allow_maginot_strategic_defence_command_platform"	desc = "maginot_strategic_defence_command_platform_desc" }
	}
	modifier = {
		ship_home_territory_fire_rate_mult = 0.50
		shipclass_starbase_damage_mult = 0.25
		shipclass_military_station_damage_mult = 0.25
		starbase_defense_platform_capacity_mult = 0.1
		army_defense_health_mult = 1.00
		army_defense_morale_mult = 1.00
	}

	potential = {
		has_gigastructural_constructs = yes
	}

	weight_modifier = {
		modifier = { factor = 0 has_global_flag = svalinn_disabled }
		modifier = { factor = 0 has_gigastructural_constructs = no }
		# voidborn as it is an interstellar emplacement
		modifier = { factor = 0 has_ascension_perk = ap_voidborn }
		# needs either unyielding or eternal vigilance
		modifier = { factor = 0
			NOR = {
				has_active_tradition = tr_unyielding_finish 
				has_active_tradition = tr_unyielding_federations_finish
				has_ascension_perk = ap_eternal_vigilance
			}
		}
		modifier = { factor = value:tech_weight_likelihood research_leader = { area = society has_trait = "leader_trait_expertise_military_theory" } }
		modifier = { factor = 2.0 is_specialist_subject_type = { TYPE = bulwark } }
		modifier = { factor = 1.5 has_ascension_perk = ap_eternal_vigilance }
		modifier = { factor = 1.5 OR = { has_active_tradition = tr_unyielding_finish has_active_tradition = tr_unyielding_federations_finish } }
		modifier = { factor = 1.5 has_ascension_perk = ap_technological_ascendancy }
		modifier = {
			factor = 2
			has_federation = yes
			federation = {
				has_federation_perk = rare_tech_boost
				any_member = { has_technology = giga_tech_svalinn_installation }
			}
		}
		# bypass: if overlord has the tech, then a bulwark can roll it regardless of conditions
		modifier = {
			add = 10
			is_specialist_subject_type = { TYPE = bulwark }
			overlord = {
				has_technology = giga_tech_svalinn_installation
			}
		}
	}

	ai_weight = {
		factor = @giga_aiweight_multiplier_strong # the AI should basically always pick this tech if it somehow managed to get requirements
		modifier = { factor = value:tech_weight_likelihood research_leader = { area = society has_trait = "leader_trait_expertise_military_theory" } }
		modifier = { factor = 1.5 	has_ascension_perk = ap_eternal_vigilance }
		modifier = { factor = 2.0 is_specialist_subject_type = { TYPE = bulwark } }
		modifier = { factor = 0 	has_global_flag = svalinn_disabled }
		modifier = { factor = 0 	has_gigastructural_constructs = no }
		# AI only has to have one of these because their AP picks are awful
		modifier = { factor = 0
			NOR = {
				has_active_tradition = tr_unyielding_finish 
				has_active_tradition = tr_unyielding_federations_finish
				has_ascension_perk = ap_eternal_vigilance
				has_ascension_perk = ap_voidborn
			}
		}
		# bypass: if overlord has the tech, then a bulwark can roll it regardless of conditions
		modifier = {
			add = 10
			is_specialist_subject_type = { TYPE = bulwark }
			overlord = {
				has_technology = giga_tech_svalinn_installation
			}
		}
	}
}
