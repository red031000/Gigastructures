				check_variable = {
					which = giga_svalinn_num_viable_neighbor_stars
					value > 0
				}
				any_neighbor_system = {
					has_star_flag = giga_svalinn_neighbor_system_to_@PREV
					check_variable = {
						which = giga_svalinn_neighbor_system_id
						value = 1
					}
				}
			}
		}
		hidden_effect = {
			event_target:giga_svalinn_construction_system = {
				random_neighbor_system = {
					limit = {
						check_variable = {
							which = giga_svalinn_neighbor_system_id
							value = 1
						}
					}
					set_star_flag = giga_svalinn_interstellar_target_system_from_@PREV
					save_event_target_as = giga_svalinn_interstellar_target_system
				}
			}

	# Cancel
	option = {
		name = "giga_svalinn_construction.011.cancel"
		custom_gui = "ehof_mainui_option_big"
		hidden_effect = {
			event_target:giga_svalinn_construction_system = {
				spawn_megastructure = {
					type = svalinn_platform_0
					owner = root
					coords_from = event_target:giga_svalinn_megastructure
					planet = event_target:giga_svalinn_megastructure.planet
				}
				set_star_flag = giga_svalinn_canceled_upgrade
			}
			event_target:giga_svalinn_megastructure = {
				remove_megastructure = this
			}
		}
	}

	after = {
		# if you didn't cancel, upgrade the mega
		event_target:giga_svalinn_construction_system = {
			if = {
				limit = { NOT = { has_star_flag = giga_svalinn_canceled_upgrade } }
				event_target:giga_svalinn_megastructure = {
					upgrade_megastructure_to = svalinn_interstellar_1	
				}
			}
			else = {
				# do nothing, but clean up the flag
				remove_star_flag = giga_svalinn_canceled_upgrade
			}
		}
	}
